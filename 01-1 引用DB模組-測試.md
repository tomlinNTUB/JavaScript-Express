### (1) asyncDB.js

``` js
'use strict';

//引用mysql模組
var mysql = require('mysql');

//建立資料庫連接池
var pool  = mysql.createPool({
    user: 'root',
    password: 'mysql',
    host: '127.0.0.1',
    database: 'north'     
});

//產生可同步執行query物件的函式
function query(sql, value) {
    return new Promise((resolve, reject) => {
        pool.query(sql, value, function (error, results, fields) {
	        if (error){
                reject(error);
            }else{
                resolve(results);
            }
        });
    });
}

//匯出可同步執行的query
module.exports = query;
```



### (2) listAllProducts.js  
``` js
'use strict';

//引用可同步執行query的物件
const query = require('./asyncDB.js');

//同步執行資料庫動作的函式
var fetchAllProducts = async function(){
    var result={};
	
    await query('SELECT * FROM product')
        .then((data) => {
            result = {code:0, data:data};  
        }, (error) => {
            result = {code:-1};
        });
		
    return result;
}

//匯出可同步執行的query
module.exports = fetchAllProducts;
```
