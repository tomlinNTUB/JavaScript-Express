### (1) asyncDB.js

``` js
'use strict';

//引用mysql模組
var mysql = require('mysql');

//建立資料庫連接池
var pool  = mysql.createPool({
    user: 'root',
    password: 'mysql',
    host: '127.0.0.1',
    database: 'north'     
});

//產生可同步執行query物件的函式
function query(sql, value) {
    return new Promise((resolve, reject) => {
        pool.query(sql, value, function (error, results, fields) {
            if (error){
                reject(error);
            }else{
                resolve(results);
            }
        });
    });
}

//匯出
module.exports = query;
```



### (2) products.js  
``` js
'use strict';

//引用操作資料庫的物件
const query = require('./asyncDB');

//------------------------------------------
/ 新增產品資料
//------------------------------------------
var addProduct = async function(proNo, proName, typNo, price){
    var result;

    await query('INSERT INTO product (proNo, proName, typNo, price) VALUES (?, ?, ?, ?)', [proNo, proName, typNo, price])
        .then((data) => {
            result = 0;  
        }, (error) => {
            result = -1;
        });
		
    return result;
}

//------------------------------------------
// 取出型態資料
//------------------------------------------
var fetchProType = async function(){
    var result;

    await query('SELECT * FROM proType')
        .then((data) => {
            result = {code:0, data:data};  
        }, (error) => {
            result = {code:-1};
        });
		
    return result;
}

//匯出
module.exports = {addProduct, fetchProType};
```


### (3) addProductForm.js

``` js
var express = require('express');
var router = express.Router();

//增加引用函式
const products = require('./utility/products');

//接收GET請求
router.get('/', function(req, res, next) {
    products.fetchProType().then(d => {
        if (d.code==0){
            res.render('addProductForm', {items:d.data});  //轉至新增頁面
        }else{
            res.render('addFail');     //導向錯誤頁面
        }  
    });
});

module.exports = router;
```

### (4) addProduct.js

``` js
var express = require('express');
var router = express.Router();

//增加引用函式
const {addProduct} = require('./utility/products');

//接收POST請求
router.post('/', function(req, res, next) {
    var proNo = req.body.proNo;          //取得傳入產品編號
    var proName = req.body.proName;      //取得傳入產品名稱
    var typNo = req.body.typNo;          //取得型態編號
    var price = Number(req.body.price);  //取得傳入價格

    addProduct(proNo, proName, typNo, price).then(d => {
        if (d==0){
            res.render('addSuccess');  //傳至成功頁面
        }else{
            res.render('addFail');     //導向錯誤頁面
        }  
    })
});

module.exports = router;
```
